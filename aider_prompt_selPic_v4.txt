### 1. ImageLoader.swift 새로 생성
PHAsset?을 받아서 이미지를 표시하는 SwiftUI View.
```swift
import SwiftUI
import Photos

struct ImageLoader: View {
    let asset: PHAsset?
    @State private var uiImage: UIImage?
    
    var body: some View {
        Group {
            if let uiImage = uiImage {
                Image(uiImage: uiImage)
                    .resizable()
                    .aspectRatio(contentMode: .fill)
                    .frame(maxWidth: .infinity, maxHeight: .infinity)
                    .clipped()
            } else if let asset = asset {
                ProgressView()
                    .progressViewStyle(.circular)
                    .tint(.white)
                    .onAppear {
                        loadImage(with: asset)
                    }
            } else {
                Image(systemName: "photo.on.rectangle")
                    .resizable()
                    .aspectRatio(contentMode: .fit)
                    .frame(width: 100, height: 100)
                    .foregroundColor(.gray)
                    .padding(40)
            }
        }
    }
    
    private func loadImage(with asset: PHAsset) {
        let options = PHImageRequestOptions()
        options.isSynchronous = false
        options.deliveryMode = .highQualityFormat
        PHImageManager.default().requestImage(
            for: asset,
            targetSize: PHImageManagerMaximumSize,
            contentMode: .aspectFill,
            options: options
        ) { image, _ in
            DispatchQueue.main.async {
                self.uiImage = image
            }
        }
    }
}
```

### 2. 앨범 내보내기 기능 (가장 중요)
ContentView.swift에 추가:
- 현재 선택된 앨범이 있는지 확인
- 선택된 앨범의 모든 PhotoItem들을 PHAsset에서 UIImage로 변환
- 각 UIImage를 jpeg Data(80% 품질)로 변환
- 임시 디렉토리에 각 사진 저장: photo_0.jpg, photo_1.jpg 등
- ZIP 파일로 압축: FileManager.createArchive(at: tempURL, items: [URL]...)
- ShareSheet로 내보내기: URL을 ShareSheet에 전달하는 sheet 표시

```swift
// MARK: - 앨범 내보내기
func exportAlbum() {
    guard let albumName = selectedAlbum,
          let photoIds = AlbumManager.shared.albumPhotos[albumName] else {
        showBannerMessage("선택된 앨범이 없습니다.")
        return
    }
    
    // Create temp directory
    let tempDir = FileManager.default.temporaryDirectory.appendingPathComponent("export_" + UUID().uuidString)
    try? FileManager.default.createDirectory(at: tempDir, withIntermediateDirectories: true)
    
    var imageURLs: [URL] = []
    let photos = photos  // local photos array
    
    for (index, photoId) in photoIds.enumerated() {
        // Find photo by ID
        if let photo = photos.first(where: { $0.id == photoId }) {
            if let asset = photo.resolveAsset() {
                let options = PHImageRequestOptions()
                options.isSynchronous = false
                options.deliveryMode = .highQualityFormat
                PHImageManager.default().requestImage(for: asset, targetSize: PHImageManagerMaximumSize, contentMode: .aspectFill, options: options) { image, _ in
                    if let image = image {
                        let jpegData = image.jpegData(compressionQuality: 0.8)
                        if let data = jpegData {
                            let fileURL = tempDir.appendingPathComponent("photo_\(index).jpg")
                            try? data.write(to: fileURL)
                            imageURLs.append(fileURL)
                            
                            // When all are done, create ZIP and share
                            if imageURLs.count == photoIds.count {
                                createAndShareArchive(at: tempDir, files: imageURLs, albumName: albumName)
                            }
                        }
                    }
                }
            }
        }
    }
}

private func createAndShareArchive(at directory: URL, files: [URL], albumName: String) {
    let archiveURL = directory.appendingPathComponent("\(albumName).zip")
    do {
        try FileManager.default.archiveEntryPaths(at: directory)
        // Actually need proper ZIP creation
        try FileManager.default.createArchive(at: archiveURL, withArchiveDirectory: directory)
        shareArchive(at: archiveURL)
    } catch {
        showBannerMessage("앨범 내보내기 실패: \(error.localizedDescription)")
    }
}

private func shareArchive(at url: URL) {
    sharingURL = url
    showShareSheet = true
}

// MARK: - ShareSheet
@State private var sharingURL: URL?
@State private var showShareSheet = false

// In body:
.sheet(isPresented: $showShareSheet) {
    ShareSheet(activityItems: [sharingURL!])
}

// MARK: - ZIP archive helper
extension FileManager {
    func createArchive(at archiveURL: URL, withArchiveDirectory directory: URL) throws {
        let fileCoordinator = NSFileCoordinator()
        fileCoordinator.coordinate(writingTo: archiveURL, error: nil) { _ in
            // Use zip command line tool
            let task = Process()
            task.executableURL = URL(fileURLWithPath: "/usr/bin/zip")
            task.arguments = ["-r", archiveURL.path, directory.path]
            try task.run()
            task.waitUntilExit()
        }
    }
}
```

### 3. UI 개선
- bannerMessage (상단 배너) - toastMessage 대신 사용
- 사진 위 더블탭 체크마크 표시

```swift
@State private var bannerMessage = ""
@State private var showBanner = false
@State private var doubleTapSelected = false

// 상단 배너 (toastMessage 대신)
if showBanner {
    Text(bannerMessage)
        .font(.headline)
        .foregroundColor(.white)
        .padding()
        .background(.black.opacity(0.7))
        .cornerRadius(10)
        .padding(.top, 50)
        .transition(.move(edge: .top))
        .id(bannerMessage)
        .onAppear {
            Timer.scheduledTimer(with: 2.0, repeats: false) { _ in
                withAnimation { showBanner = false }
            }
        }
}

// 더블탭 체크마크
ImageLoader(asset: photos[currentIndex].resolveAsset())
    .onTapGesture(count: 2) {
        doubleTapSelected.toggle()
        showBanner("선택" + (doubleTapSelected ? "됨" : "해제"))
    }
```
